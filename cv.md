function deepCount(a){
    let counter = 0;

    for(let i = 0; i<a.length; i++){
        if(typeof a[i] !== 'object'){
            counter++;
        }else if(typeof a[i] == 'object'){
            counter++
            counter = counter + deepCount(a[i]);
        }
    }
